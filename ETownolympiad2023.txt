#include <iostream>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    int n,m,i,ii,iii;
    bool flag=false;
    cin>>n>>m;
    string a[m][n];
    for(i=0;i<n;i++)
    {
        for(ii=0;ii<m;ii++)
        {
            a[ii][i]=".";
        }
    }
    for(i=0,ii=0,iii=m-1;i<n;i++)
    {
        if(ii==m)
        {
            ii=ii-2;
            iii=iii+2;
            flag=true;
        }
        if(ii==-1)
        {
            ii=ii+2;
            iii=iii-2;
            flag=false;
        }
        a[ii][i]="x";
        a[iii][i]="x";
        if(flag==true)
        {
            ii--;
            iii++;
        }
        else
        {
            ii++;
            iii--;
        }
    }
    for(i=0;i<n;i++)
    {
        for(ii=0;ii<m;ii++)
        {
            cout<<a[ii][i];
        }
        cout<<endl;
    }
    return 0;
}
