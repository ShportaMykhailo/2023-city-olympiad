умова
Герой олімпіад Василько отримав у подарунок перестановку N чисел a1,a2,…,aN. 
(у перестановці кожне з чисел 1, 2,…,N міститься рівно один раз) Він застосував її для набору чисел 1,2,…,N таким чином. 
Перше число (тобто 1) поставив на позицію a1, друге (тобто 2) - на a2, останнє (тобто N) - на aN. 
Наприклад, для перестановки 2 4 5 1 3 Василько матиме 4 1 5 2 3. Далі Василько повторює ці дії (i те число ставиться на ai позицію) для отриманого набору чисел. 
Так, з 4 1 5 2 3 вийде 2 4 3 1 5. Скільки дій знадобиться Васильку, щоб знову отримати 1, 2,…,N?

код
#include <iostream>

using namespace std;

int main()
{
    int i,n,seq;
    bool flag;
    seq=1;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    int b[n];
    for(i=0;i<n;i++)
    {
        b[i]=i+1;
    }
    int c[n];
    int m=2;
    for(i=3;i<=n;i++)
    {
        m=i*m;
    }
    while(seq<m)
    {
        for(i=0;i<n;i++)
        {
            c[a[i]-1]=b[i];
        }
        seq++;
        for(i=0;i<n;i++)
        {
            if(a[i]==c[i])
            {
                flag=true;
            }
            else
            {
                flag=false;
                break;
            }
        }
            if(flag==true)
            {
                cout<<seq;
                return 0;
            }
        for(i=0;i<n;i++)
        {
            b[i]=c[i];
        }
    }
    return 0;
}
